# boot LetuxOS on Odroid-M1

# Bootscript using the new unified bootcmd handling
#
# Expects to be called with the following environment variables set:
#
#  devtype              e.g. mmc/scsi etc
#  devnum               The device number of the given type
#  bootpart             The partition containing the boot files
#  prefix               Prefix within the boot partition to the boot files
#  kernel_addr_r        Address to load the kernel to
#  fdt_addr_r           Address to load the FDT to
#  ramdisk_addr_r       Address to load the initrd to.
#
# The uboot must support the booti and generic filesystem load commands.

printenv

setenv board odroid${variant}

setenv bootargs " ${bootargs} root=UUID=795e9814-c53e-4f94-b71e-13c33c2196f6 quiet splash plymouth.ignore-serial-consoles"
setenv overlay_resize 8192

setenv bootlabel "Letux OS"

# Default serial console
setenv console "ttyFIQ0"

# Default TTY console
setenv bootargs "${bootargs} earlycon=uart8250,mmio32,0xfe660000 console=tty1"

# MISC
#
setenv bootargs "${bootargs} pci=nomsi"
setenv bootargs "${bootargs} fsck.mode=force fsck.repair=yes"
setenv bootargs "${bootargs} mtdparts=sfc_nor:0x20000@0xe0000(env),0x200000@0x100000(uboot),0x100000@0x300000(splash),0xc00000@0x400000(firmware)"

setenv fdtfile "rk3568-odroid-m1.dtb"

if test -n "${console}"; then
  setenv bootargs "${bootargs} console=${console}"
fi

if test -z "${distro_bootpart}"; then
  setenv partition ${bootpart}
else
  setenv partition ${distro_bootpart}
fi

# load device tree

load ${devtype} ${devnum}:${partition} ${fdt_addr_r} ${prefix}rockchip/${fdtfile}
fdt addr ${fdt_addr_r}

# load kernel without initrd...

load ${devtype} ${devnum}:${partition} ${ramdisk_addr_r} ${prefix}Image.gz \
&& unzip ${ramdisk_addr_r} ${kernel_addr_r} \
&& echo "Booting Debian ${fk_kvers} from ${devtype} ${devnum}:${partition}..." \
&& booti ${kernel_addr_r} - ${fdt_addr_r}

exit

load ${devtype} ${devnum}:${partition} ${ramdisk_addr_r} ${prefix}Image.gz \
&& unzip ${ramdisk_addr_r} ${kernel_addr_r} \
&& load ${devtype} ${devnum}:${partition} ${ramdisk_addr_r} ${prefix}initrd.img-${fk_kvers} \
&& echo "Booting Debian ${fk_kvers} from ${devtype} ${devnum}:${partition}..." \
&& booti ${kernel_addr_r} ${ramdisk_addr_r} ${fdt_addr_r}

exit


# stuff from pinephone

part uuid ${devtype} ${devnum}:${distro_bootpart} uuid
setenv console ttyS0,115200n8
setenv bootargs console=${console} rootfstype=ext4,ext3 root=PARTUUID=${uuid} rw rootwait ignore_loglevel
setenv kernel_addr_z 0x44080000

if load ${devtype} ${devnum}:${distro_bootpart} ${kernel_addr_z} /boot/Image.gz; then
  unzip ${kernel_addr_z} ${kernel_addr_r}
  if load ${devtype} ${devnum}:${distro_bootpart} ${fdt_addr_r} /boot/dtb/allwinner/sun50i-a64-pinephone-1.1.dtb; then
    if load ${devtype} ${devnum}:${distro_bootpart} ${ramdisk_addr_r} /boot/initrd.img; then
      booti ${kernel_addr_r} ${ramdisk_addr_r}:${filesize} ${fdt_addr_r};
    else
      booti ${kernel_addr_r} - ${fdt_addr_r};
    fi;
  else if load ${devtype} ${devnum}:${distro_bootpart} ${fdt_addr_r} /boot/dtb/allwinner/sun50i-a64-pinephone.dtb; then
    if load ${devtype} ${devnum}:${distro_bootpart} ${ramdisk_addr_r} /boot/initrd.img; then
      booti ${kernel_addr_r} ${ramdisk_addr_r}:${filesize} ${fdt_addr_r};
    else
      booti ${kernel_addr_r} - ${fdt_addr_r};
    fi;
  fi;
fi
